```javascript
// 每次修改后运行验证
function validateQuestion(q) {
  const required = ['id', 'category', 'type', 'content', 'answer'];
  for (const field of required) {
    if (!q[field]) throw new Error(`Missing field: ${field}`);
  }

  // 题型特定验证
  if (q.type === 'single_choice') {
    if (!q.content.options || q.content.options.length < 2) {
      throw new Error('选择题至少需要2个选项');
    }
  }

  return true;
}
```

```vue
<template>
  <div class="question-display">
    <!-- 题目头部 -->
    <div class="question-header">
      <span class="category">{{ question.category.primary }}</span>
      <span class="type">{{ typeLabel }}</span>
      <span class="difficulty" :class="question.difficulty">

      </span>
    </div>

    <!-- 复合题 -->
    <div v-if="question.type === 'composite'" class="composite-question">
      <div class="material" v-if="question.content.material">
        {{ question.content.material }}
      </div>
      <div class="sub-questions">
        <div 
          v-for="sub in question.content.subQuestions" 
          :key="sub.id"
          class="sub-question"
        >
          <span class="order">({{ sub.order }})</span>
          <span class="sub-type">[{{ sub.type }}]</span>
          <p>{{ sub.stem }}</p>
        </div>
      </div>
    </div>

    <!-- 普通题 -->
    <div v-else class="normal-question">
      <p class="stem">{{ question.content.stem }}</p>

      <!-- 选择题选项 -->
      <div v-if="question.content.options" class="options">
        <div 
          v-for="opt in question.content.options" 
          :key="opt.label"
          class="option"
        >
          <span class="label">{{ opt.label }}.</span>
          <span class="text">{{ opt.text }}</span>
        </div>
      </div>
    </div>

    <!-- 答案区域 -->
    <div class="answer-section" v-if="showAnswer">
      <p><strong>答案:</strong> {{ question.answer.correct }}</p>

        <strong>解析:</strong> {{ question.answer.analysis }}

<script>

  props: ['question', 'showAnswer'],
  computed: {
    typeLabel() {
      const map = {
        'single_choice': '单选题',
        'multiple_choice': '多选题',
        'fill_blank': '填空题',
        'short_answer': '简答题',
        'analysis': '分析题',
        'essay': '作文题',
        'composite': '复合题',
        'language_expr': '语言表达'
      };
      return map[this.question.type] || this.question.type;
    },
    difficultyLabel() {
      const map = {
        'easy': '易',
        'medium': '中',
        'hard': '难'
      };
      return map[this.question.difficulty] || this.question.difficulty;
    }
  }
};
</script>
```

### 按分类筛选题目

```javascript
// 筛选函数
function filterQuestions(questions, filters) {
  return questions.filter(q => {
    // 按知识分类筛选
    if (filters.category && q.category.primary !== filters.category) {
      return false;
    }

    // 按题型筛选
    if (filters.type && q.type !== filters.type) {
      return false;
    }

    // 按难度筛选
    if (filters.difficulty && q.difficulty !== filters.difficulty) {
      return false;
    }

    // 按标签筛选
    if (filters.tag && !q.tags.includes(filters.tag)) {
      return false;
    }

    return true;
  });
}

// 使用示例
const filtered = filterQuestions(questions, {
  category: '人物形象',
  type: 'analysis',
  difficulty: 'hard'
});
```
